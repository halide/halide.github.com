157size_t StringTable::parse(void *user_context, const char *str, const char *delim) {
158 if (StringUtils::is_empty(str)) {
159 return 0;
160 }
161
162 size_t delim_length = strlen(delim);
163 size_t total_length = strlen(str);
164 size_t entry_count = StringUtils::count_tokens(str, delim);
165 if (entry_count < 1) {
166 return 0;
167 }
168
169 resize(user_context, entry_count);
170
171 // save each entry into the table
172 size_t index = 0;
173 const char *ptr = str;
174 while (!StringUtils::is_empty(ptr) && (index < entry_count)) {
175 size_t ptr_offset = ptr - str;
176 const char *next_delim = strstr(ptr, delim);
177 size_t token_length = (next_delim == nullptr) ? (total_length - ptr_offset) : (next_delim - ptr);
178 if (token_length > 0 && index < contents.size()) {
179 StringStorage *storage_ptr = static_cast<StringStorage *>(contents[index]);
180 storage_ptr->assign(user_context, ptr, token_length);
181 pointers.assign(user_context, index, storage_ptr->data());
182 ++index;
183 }
184 ptr = (next_delim != nullptr) ? (next_delim + delim_length) : nullptr;
185 }
186 return entry_count;
187}
