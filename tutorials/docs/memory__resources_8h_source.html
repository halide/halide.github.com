31// Hint for allocation requests indicating intended update
32// frequency for modifying the contents of the allocation
33enum class MemoryUsage {
34 InvalidUsage, //< invalid enum value
35 StaticStorage, //< intended for static storage, whereby the contents will be set once and remain unchanged
36 DynamicStorage, //< intended for dyanmic storage, whereby the contents will be set frequently and change constantly
37 UniformStorage, //< intended for fast & small fixed read-only uniform storage (intended for passing shader parameters), whereby the contents will be set once and remain unchanged
38 TransferSrc, //< intended for staging storage updates, whereby the contents will be used as the source of a transfer
39 TransferDst, //< intended for staging storage updates, whereby the contents will be used as the destination of a transfer
40 TransferSrcDst, //< intended for staging storage updates, whereby the contents will be used either as a source or destination of a transfer
41 DefaultUsage //< default usage (use any valid usage -- unable to determine prior to usage)
42};

108// Returns an aligned byte offset to adjust the given offset based on alignment constraints
109// -- Alignment must be power of two!
110ALWAYS_INLINE size_t aligned_offset(size_t offset, size_t alignment) {
111 return (offset + (alignment - 1)) & ~(alignment - 1);
112}
113
114// Returns a padded size to accomodate an adjusted offset due to alignment constraints
115// -- Alignment must be power of two!
116ALWAYS_INLINE size_t aligned_size(size_t offset, size_t size, size_t alignment) {
117 size_t actual_offset = aligned_offset(offset, alignment);
118 size_t padding = actual_offset - offset;
119 size_t actual_size = padding + size;
120 return actual_size;
121}

123// Clamps the given value to be within the [min_value, max_value] range
124ALWAYS_INLINE size_t clamped_size(size_t value, size_t min_value, size_t max_value) {
125 size_t result = (value < min_value) ? min_value : value;
126 return (result > max_value) ? max_value : result;
127}
