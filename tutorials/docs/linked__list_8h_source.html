133LinkedList::prepend(void *user_context) {

135 if (empty()) {
136 front_ptr = entry_ptr;
137 back_ptr = entry_ptr;

202void LinkedList::clear(void *user_context) {

216void LinkedList::remove(void *user_context, EntryType *entry_ptr) {
217 halide_abort_if_false(user_context, (entry_ptr != nullptr));
218 halide_abort_if_false(user_context, (entry_count > 0));
219
220 if (entry_ptr->prev_ptr != nullptr) {
221 entry_ptr->prev_ptr->next_ptr = entry_ptr->next_ptr;
222 } else {
223 halide_abort_if_false(user_context, (front_ptr == entry_ptr));
224 front_ptr = entry_ptr->next_ptr;
225 }
226
227 if (entry_ptr->next_ptr != nullptr) {
228 entry_ptr->next_ptr->prev_ptr = entry_ptr->prev_ptr;
229 } else {
230 halide_abort_if_false(user_context, (back_ptr == entry_ptr));
231 back_ptr = entry_ptr->prev_ptr;
232 }
233
234 reclaim(user_context, entry_ptr);
235 --entry_count;
236}
