180void PointerTable::resize(void *user_context, size_t entry_count, bool realloc) {
181 size_t current_size = capacity;
182 size_t requested_size = entry_count;
183 size_t minimum_size = default_capacity;
184 size_t actual_size = current_size;
185 count = requested_size;
186
187#ifdef DEBUG_RUNTIME
188 debug(user_context) << "PointerTable: Resize ("
189 << "requested_size=" << (int32_t)requested_size << " "
190 << "current_size=" << (int32_t)current_size << " "
191 << "minimum_size=" << (int32_t)minimum_size << " "
192 << "sizeof(void*)=" << (int32_t)sizeof(void *) << " "
193 << "realloc=" << (realloc ? "true" : "false") << ")...\n";
194#endif
195
196 // increase capacity upto 1.5x existing (or at least min_capacity)
197 if (requested_size > current_size) {
198 actual_size = max(requested_size, max(current_size * 3 / 2, minimum_size));
199 } else if (!realloc) {
200 return;
201 }
202
203 allocate(user_context, actual_size);
204}

251void PointerTable::replace(void *user_context, size_t index, const void **array, size_t array_size) {
252 halide_debug_assert(user_context, index < count);
253 size_t remaining = count - index;
254 size_t copy_count = min(remaining, array_size);

271void PointerTable::insert(void *user_context, size_t index, const void **array, size_t array_size) {
272 halide_debug_assert(user_context, index <= count);
273 const size_t last_index = size();
274 resize(user_context, last_index + array_size);
275 if (index < last_index) {
276 size_t src_offset = index * sizeof(void *);
277 size_t dst_offset = (index + array_size) * sizeof(void *);
278 size_t bytes = (last_index - index) * sizeof(void *);
279 memmove(ptr + dst_offset, ptr + src_offset, bytes);
280 }
281 replace(user_context, index, array, array_size);
282}

330void PointerTable::allocate(void *user_context, size_t new_capacity) {
331 if (new_capacity != capacity) {
332 halide_abort_if_false(user_context, allocator.allocate != nullptr);
333 size_t bytes = new_capacity * sizeof(void *);
334
335#ifdef DEBUG_RUNTIME
336 debug(user_context) << "PointerTable: Allocating (bytes=" << (int32_t)bytes << " allocator=" << (void *)allocator.allocate << ")...\n";
337#endif
338
339 void *new_ptr = bytes ? allocator.allocate(user_context, bytes) : nullptr;
340 if (count != 0 && ptr != nullptr && new_ptr != nullptr) {
341 memcpy(new_ptr, ptr, count * sizeof(void *));
342 }
343 if (ptr != nullptr) {
344 halide_abort_if_false(user_context, allocator.deallocate != nullptr);
345 allocator.deallocate(user_context, ptr);
346 }
347 capacity = new_capacity;
348 ptr = static_cast<void **>(new_ptr);
349 }
