00025 /** The abstract base classes for a node in the Halide IR. */
00026 struct IRNode {
00027 
00028     /** We use the visitor pattern to traverse IR nodes throughout the
00029      * compiler, so we have a virtual accept method which accepts
00030      * visitors.
00031      */
00032     virtual void accept(IRVisitor *v) const = 0;
00033     IRNode() {}
00034     virtual ~IRNode() {}
00035 
00036     /** These classes are all managed with intrusive reference
00037        counting, so we also track a reference count. It's mutable
00038        so that we can do reference counting even through const
00039        references to IR nodes. */
00040     mutable RefCount ref_count;
00041 
00042     /** Each IR node subclass should return some unique pointer. We
00043      * can compare these pointers to do runtime type
00044      * identification. We don't compile with rtti because that
00045      * injects run-time type identification stuff everywhere (and
00046      * often breaks when linking external libraries compiled
00047      * without it), and we only want it for IR nodes. */
00048     virtual const IRNodeType *type_info() const = 0;
00049 };
