672 /** Check if a call node is pure within a pipeline, meaning that
673 * the same args always give the same result, and the calls can be
674 * reordered, duplicated, unified, etc without changing the
675 * meaning of anything. Not transitive - doesn't guarantee the
676 * args themselves are pure. An example of a pure Call node is
677 * sqrt. If in doubt, don't mark a Call node as pure. */
678 bool is_pure() const {
679 return (call_type == PureExtern ||
680 call_type == Image ||
681 call_type == PureIntrinsic);
682 }
683
684 bool is_intrinsic() const {
685 return (call_type == Intrinsic ||
686 call_type == PureIntrinsic);
687 }
688
689 bool is_intrinsic(IntrinsicOp op) const {
690 return is_intrinsic() && this->name == get_intrinsic_name(op);
691 }
692
693 bool is_intrinsic(std::initializer_list<IntrinsicOp> intrinsics) const {
694 for (IntrinsicOp i : intrinsics) {
695 if (is_intrinsic(i)) {
696 return true;
697 }
698 }
699 return false;
700 }
701
702 bool is_tag() const {
703 return is_intrinsic({Call::likely, Call::likely_if_innermost, Call::strict_float});

706 /** Returns a pointer to a call node if the expression is a call to
707 * one of the requested intrinsics. */
708 static const Call *as_intrinsic(const Expr &e, std::initializer_list<IntrinsicOp> intrinsics) {
709 if (const Call *c = e.as<Call>()) {
710 for (IntrinsicOp i : intrinsics) {
711 if (c->is_intrinsic(i)) {
712 return c;
713 }
714 }
715 }
716 return nullptr;
717 }
