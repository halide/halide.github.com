101 /** A helper function for mutator-like things to mutate regions */
102 template<typename Mutator, typename... Args>
103 std::pair<Region, bool> mutate_region(Mutator *mutator, const Region &bounds, Args&&... args) {
104  Region new_bounds(bounds.size());
105  bool bounds_changed = false;
106 
107  for (size_t i = 0; i < bounds.size(); i++) {
108  Expr old_min = bounds[i].min;
109  Expr old_extent = bounds[i].extent;
110  Expr new_min = mutator->mutate(old_min, std::forward<Args>(args)...);
111  Expr new_extent = mutator->mutate(old_extent, std::forward<Args>(args)...);
112  if (!new_min.same_as(old_min)) {
113  bounds_changed = true;
114  }
115  if (!new_extent.same_as(old_extent)) {
116  bounds_changed = true;
117  }
118  new_bounds[i] = Range(new_min, new_extent);
119  }
120  return {new_bounds, bounds_changed};
121 }
