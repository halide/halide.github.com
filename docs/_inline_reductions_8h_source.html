13 /** An inline reduction. This is suitable for convolution-type
14  * operations - the reduction will be computed in the innermost loop
15  * that it is used in. The argument may contain free or implicit
16  * variables, and must refer to some reduction domain. The free
17  * variables are still free in the return value, but the reduction
18  * domain is captured - the result expression does not refer to a
19  * reduction domain and can be used in a pure function definition.
20  *
21  * An example using \ref sum :
22  *
23  \code
24  Func f, g;
25  Var x;
26  RDom r(0, 10);
27  f(x) = x*x;
28  g(x) = sum(f(x + r));
29  \endcode
30  *
31  * Here g computes some blur of x, but g is still a pure function. The
32  * sum is being computed by an anonymous reduction function that is
33  * scheduled innermost within g.
34  */
35 //@{
36 EXPORT Expr sum(Expr, const std::string &s = "sum");
37 EXPORT Expr product(Expr, const std::string &s = "product");
38 EXPORT Expr maximum(Expr, const std::string &s = "maximum");
39 EXPORT Expr minimum(Expr, const std::string &s = "minimum");
40 //@}
